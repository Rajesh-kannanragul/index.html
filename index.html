Connect to Supabase

Follow these steps once to connect your real database. Takes about 5 minutes.

📋 Step-by-step Supabase setup

1
Go to supabase.com → Create a free account → New project → choose a name and password
2
In your project → Settings → API → copy your Project URL and anon public key
3
Go to SQL Editor → paste and run the SQL below to create the 3 tables
4
Paste your URL and key below and click Connect

📄 SQL to run in Supabase SQL Editor

-- Run this in Supabase SQL Editor

-- 1. Budgets table
create table if not exists budgets (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade not null,
  year int not null,
  month int not null,
  amount numeric not null,
  created_at timestamptz default now(),
  unique(user_id, year, month)
);
alter table budgets enable row level security;
create policy "Users manage own budgets" on budgets
  for all using (auth.uid() = user_id);

-- 2. Transactions table
create table if not exists transactions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade not null,
  year int not null,
  month int not null,
  day int not null,
  amount numeric not null,
  category text not null,
  description text,
  tx_time text,
  created_at timestamptz default now()
);
alter table transactions enable row level security;
create policy "Users manage own transactions" on transactions
  for all using (auth.uid() = user_id);

-- 3. Profiles table
create table if not exists profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  full_name text,
  created_at timestamptz default now()
);
alter table profiles enable row level security;
create policy "Users manage own profile" on profiles
  for all using (auth.uid() = id);

-- Auto-create profile on signup
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer set search_path = public as $$
begin
  insert into public.profiles (id, full_name)
  values (new.id, new.raw_user_meta_data->>'full_name');
  return new;
end;
$$;
drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();

-- 5. Alert logs table (audit trail of every email notification sent)
create table if not exists alert_logs (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade not null,
  year int not null,
  month int not null,
  period int not null,
  alert_type text not null,
  message text not null,
  email_sent_to text,
  created_at timestamptz default now()
);
alter table alert_logs enable row level security;
create policy "Users view own alert logs" on alert_logs
  for select using (auth.uid() = user_id);
create policy "Users insert own alert logs" on alert_logs
  for insert with check (auth.uid() = user_id);

Enable email alerts (optional but recommended)

To send real budget alert emails to users, deploy a Supabase Edge Function called send-budget-alert. It receives the alert details (orange/yellow/red) and emails the user via a provider like Resend or SendGrid.

// supabase/functions/send-budget-alert/index.ts import { serve } from "https://deno.land/std/http/server.ts"; serve(async (req) => { const { to, name, alertType, message, month, year, period, spent, limit, over, pctOver } = await req.json(); const subject = `${alertType.toUpperCase()} Alert — Budget Detector (${month} ${year})`; const html = ` <h2>${alertType === 'red' ? '🚨' : alertType === 'yellow' ? '🟡' : '⚠️'} ${message}</h2> <p>Period ${period} of ${month} ${year}</p> <p>Spent: ₹${spent} / Limit: ₹${limit} (₹${over} over, ${pctOver}%)</p> `; // Send via Resend (resend.com) — replace with your API key await fetch("https://api.resend.com/emails", { method: "POST", headers: { "Authorization": `Bearer ${Deno.env.get("RESEND_API_KEY")}`, "Content-Type": "application/json" }, body: JSON.stringify({ from: "Budget Detector <alerts@yourdomain.com>", to, subject, html }) }); return new Response(JSON.stringify({ ok: true }), { headers: { "Content-Type": "application/json" } }); });

Deploy with: supabase functions deploy send-budget-alert
Without this function deployed, the app still shows alerts inside the dashboard — emails are simply skipped silently.
